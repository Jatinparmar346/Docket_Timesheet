function renderDateDisplay(){

  document.getElementById('date-display').textContent=S.currentDate.toLocaleDateString('en-US',opts);

function renderWeekGrid(){
  var cur=new Date(S.currentDate),dow=cur.getDay(),mon=new Date(cur);
  mon.setDate(cur.getDate()-(dow===0?6:dow-1));
  var days=['Mon','Tue','Wed','Thu','Fri','Sat','Sun'],html='';
  for(var i=0;i<7;i++){
    var d=new Date(mon);d.setDate(mon.getDate()+i);
    var ds=fmt(d),isToday=ds===fmt(new Date()),isSel=ds===fmt(S.currentDate);
    var hrs=myDayEntries(ds).reduce(function(s,e){return s+e.hours;},0);
    html+='<div class="week-day'+(isToday?' today':'')+(isSel?' selected':'')+'" onclick="selectDay(\''+ds+'\')">'
      +'<div class="week-day-name">'+days[i]+'</div><div class="week-day-num">'+d.getDate()+'</div>'
      +'<div class="week-day-hours">'+(hrs>0?fmtH(hrs):'—')+'</div></div>';
  }
  document.getElementById('week-grid').innerHTML=html;
}
function selectDay(ds){S.currentDate=new Date(ds+'T12:00:00');document.getElementById('entry-date').value=ds;renderAll();}

function myDayEntries(ds){
  var u=S.currentUser;
  return S.entries.filter(function(e){
    if(e.date!==ds) return false;
    if(perm().isClient) return e.client===u.clientId;
    if(perm().canAdmin||perm().canManage||perm().canApprove||u.role==='viewer') return true;
    return e.userId===u.id;
  });
}

function pauseResumeTimer(){
  if(!S.timer.running)return;
  if(!S.timer.paused){
    S.timer.paused=true;S.timer.pausedElapsed=Math.floor((Date.now()-S.timer.startTime)/1000);
    document.getElementById('pause-timer-btn').innerHTML='<svg width="12" height="12" viewBox="0 0 24 24" fill="currentColor"><polygon points="5 3 19 12 5 21 5 3"/></svg>Resume';
    document.getElementById('running-label').textContent='⏸ Paused';
    document.getElementById('running-dot').style.animationPlayState='paused';

    S.timer.paused=false;S.timer.startTime=Date.now()-S.timer.pausedElapsed*1000;

function startBreak(){
  if(S.breakTimer.running){toast('Break already running');return;}
  if(S.timer.running&&!S.timer.paused)pauseResumeTimer();
  S.breakTimer.running=true;S.breakTimer.startTime=Date.now();
  document.getElementById('break-banner').style.display='flex';
  document.getElementById('break-meta').textContent='Break started at '+new Date().toLocaleTimeString([],{hour:'2-digit',minute:'2-digit'});
  toast('☕ Break started');
}
function endBreak(){
  if(!S.breakTimer.running)return;
  var elapsed=Math.floor((Date.now()-S.breakTimer.startTime)/1000);
  S.totalBreakSecs=(S.totalBreakSecs||0)+elapsed;
  S.breakTimer.running=false;S.breakTimer.startTime=null;
  document.getElementById('break-banner').style.display='none';
  document.getElementById('break-timer-display').textContent='0:00:00';
  var tbd=document.getElementById('total-break-display');
  if(tbd&&S.totalBreakSecs>0){
    tbd.style.display='inline-flex';
    var bm=Math.floor(S.totalBreakSecs/60),bs=S.totalBreakSecs%60;
    document.getElementById('total-break-val').textContent=bm+':'+String(bs).padStart(2,'0');
  }
}
function stopTimer(){
  if(!S.timer.running)return;
  var elapsed=S.timer.paused?S.timer.pausedElapsed:(Date.now()-S.timer.startTime)/1000;
  S.timer.running=false;S.timer.paused=false;
  var hours=Math.max(elapsed/3600,0.017);
  var client=document.getElementById('entry-client').value;
  var project=document.getElementById('entry-project').value;
  var task=document.getElementById('entry-task').value;
  var desc=document.getElementById('entry-description').value;
  var dateStr=document.getElementById('entry-date').value||fmt(S.currentDate);
  var st=new Date(S.timer.startTime),et=new Date();
  var pad=function(n){return String(n).padStart(2,'0');};
  var breakMins=Math.round(S.totalBreakSecs/60);
  S.entries.push({id:Date.now(),userId:S.currentUser.id,date:dateStr,client:client,project:project,task:task,description:desc,hours:hours,start:pad(st.getHours())+':'+pad(st.getMinutes()),end:pad(et.getHours())+':'+pad(et.getMinutes()),breakMins:breakMins,approvalStatus:'pending'});
  S.timer.startTime=null;S.timer.pausedElapsed=0;S.totalBreakSecs=0;
  document.getElementById('running-banner').style.display='none';
  var btn=document.getElementById('start-stop-btn');
  btn.innerHTML='<svg width="13" height="13" viewBox="0 0 24 24" fill="currentColor"><polygon points="5 3 19 12 5 21 5 3"/></svg>Start Timer';
  btn.className='btn btn-green';
  document.getElementById('live-timer').textContent='0:00:00';
  document.getElementById('running-timer-display').textContent='0:00:00';
  renderAll();toast('Time entry saved ✓');
}
function tickTimers(){

  }
}

function getMonday(d){
  var day=d.getDay(),diff=d.getDate()-(day===0?6:day-1);
  return new Date(d.setDate(diff));
}
function submitWeeklyTimesheet(){
  var u=S.currentUser;
  var ws=fmt(getMonday(new Date()));
  var we=fmt(new Date(getMonday(new Date()).getTime()+6*86400000));
  var existing=S.timesheets.find(function(ts){return ts.userId===u.id&&ts.weekStart===ws;});
  if(existing&&existing.status==='submitted'){toast('Already submitted for this week');return;}
  if(existing&&existing.status==='approved'){toast('This week is already approved');return;}
  if(existing){
    existing.status='submitted';existing.submittedAt=new Date().toISOString();
  } else {
    S.timesheets.push({id:'ts'+Date.now(),userId:u.id,weekStart:ws,weekEnd:we,status:'submitted',submittedAt:new Date().toISOString(),reviewedBy:null,reviewedAt:null,comment:''});
  }
  toast('📤 Timesheet submitted for approval!');
}

function renderApprovals(){
  var statusF=document.getElementById('approval-filter-status').value;

  // Populate emp filter
  var aef=document.getElementById('approval-filter-emp');
  aef.innerHTML='<option value="">All Employees</option>'+USERS.filter(function(u){return u.role==='employee';}).map(function(u){return '<option value="'+u.id+'">'+u.name+'</option>';}).join('');
  if(empF) aef.value=empF;

  var list=document.getElementById('approval-list');
  // Group entries by userId+week
  var weekMap={};
  S.entries.forEach(function(e){
    var wk=getWeekStart(e.date);
    var key=e.userId+'__'+wk;
    if(!weekMap[key]) weekMap[key]={userId:e.userId,weekStart:wk,entries:[]};
    weekMap[key].entries.push(e);
  });
  // Merge with submitted timesheets
  S.timesheets.forEach(function(ts){
    var key=ts.userId+'__'+ts.weekStart;
    if(!weekMap[key]) weekMap[key]={userId:ts.userId,weekStart:ts.weekStart,entries:[]};
    weekMap[key].tsId=ts.id;
    weekMap[key].status=ts.status;
    weekMap[key].comment=ts.comment;
    weekMap[key].submittedAt=ts.submittedAt;
    weekMap[key].reviewedBy=ts.reviewedBy;
  });
  var rows=Object.values(weekMap).filter(function(r){
    if(statusF && (r.status||'draft')!==statusF) return false;
    if(empF && r.userId!==empF) return false;
    return true;
  }).sort(function(a,b){return b.weekStart.localeCompare(a.weekStart);});

  if(!rows.length){list.innerHTML='<div class="empty-state"><div class="empty-icon">📋</div><div class="empty-title">No timesheets found</div><div class="empty-sub">Adjust filters or wait for submissions</div></div>';return;}
  list.innerHTML=rows.map(function(r){
    var emp=USERS.find(function(u){return u.id===r.userId;})||{name:'Unknown',color:'#888'};
    var status=r.status||'draft';
    var totalH=r.entries.reduce(function(s,e){return s+e.hours;},0);
    var statusBadgeCls={draft:'approval-pending',submitted:'approval-submitted',approved:'approval-approved',rejected:'approval-rejected'}[status]||'approval-pending';
    var statusLabel={draft:'Draft',submitted:'Submitted',approved:'Approved',rejected:'Rejected'}[status]||status;
    var we=getWeekEnd(r.weekStart);
    return '<div class="approval-week-card">'
      +'<div class="approval-week-header">'
      +'<div style="display:flex;align-items:center;gap:10px;">'
      +'<div style="width:32px;height:32px;border-radius:50%;background:'+emp.color+';display:flex;align-items:center;justify-content:center;font-size:12px;font-weight:700;color:#fff;">'+initials(emp.name)+'</div>'
      +'<div><div style="font-weight:600;font-size:14px;">'+emp.name+'</div><div style="font-size:12px;color:var(--ink-light);">'+r.weekStart+' → '+we+'</div></div>'
      +'</div>'
      +'<div style="display:flex;align-items:center;gap:12px;">'
      +'<span style="font-size:13px;color:var(--ink-mid);">'+fmtH(totalH)+' hrs</span>'
      +'<span class="approval-status-badge '+statusBadgeCls+'">'+statusLabel+'</span>'
      +(status==='submitted'?'<button class="btn btn-sm btn-primary" onclick="openApprovalModal(\''+r.userId+'\',\''+r.weekStart+'\')">Review</button>':'')
      +'</div></div>'
      +'<div class="approval-entries">'
      +(r.entries.length?r.entries.slice(0,3).map(function(e){return '<div style="display:flex;justify-content:space-between;font-size:13px;padding:4px 0;border-bottom:1px solid var(--border);"><span style="color:var(--ink-mid);">'+e.date+' — '+(e.project||e.client||'—')+' <span style="font-size:11px;color:var(--ink-light);">'+(e.description||'')+'</span></span><span class="text-mono" style="font-weight:600;">'+fmtH(e.hours)+'</span></div>';}).join('')+'':'<span style="font-size:13px;color:var(--ink-light);">No entries this week</span>')
      +(r.entries.length>3?'<div style="font-size:12px;color:var(--ink-light);margin-top:4px;">+' +(r.entries.length-3)+' more entries</div>':'')

  }).join('');

function assignProjectToClient(clientName){
  var key=clientName.replace(/\s/g,'_');
  var sel=document.getElementById('assign-proj-'+key);
  if(!sel||!sel.value)return;
  var proj=S.projects.find(function(p){return p.id===sel.value;});
  if(proj){proj.clientId=clientName;renderManageLists();populateSelects();toast('Project assigned to '+clientName);}
}
function unassignProjectFromClient(projId,clientName){
  var proj=S.projects.find(function(p){return p.id===projId;});
  if(proj){proj.clientId=null;renderManageLists();populateSelects();toast('Project unassigned from '+clientName);}
}

function addClient(){
  var v=document.getElementById('new-client-input').value.trim();
  if(!v||S.clients.indexOf(v)>-1){toast('Client already exists or empty');return;}
  S.clients.push(v);
  document.getElementById('new-client-input').value='';
  renderManageLists();populateSelects();toast('Client "'+v+'" added');
}
function removeClient(n){
  S.clients=S.clients.filter(function(c){return c!==n;});
  // remove from projects
  S.projects.forEach(function(p){if(p.clientId===n)p.clientId=null;});
  renderManageLists();populateSelects();
}
function addProject(){
  var v=document.getElementById('new-project-input').value.trim();
  var clientId=document.getElementById('new-project-client').value||null;
  if(!v){toast('Project name required');return;}
  if(S.projects.find(function(p){return p.name===v;})){toast('Project already exists');return;}
  S.projects.push({id:'p'+Date.now(),name:v,clientId:clientId});
  document.getElementById('new-project-input').value='';
  renderManageLists();populateSelects();toast('Project "'+v+'" added');
}
function removeProject(id){
  S.projects=S.projects.filter(function(p){return p.id!==id;});
  renderManageLists();populateSelects();
}

function openInlineEdit(type, val){
  S.inlineEditType=type;S.inlineEditOldVal=val;
  document.getElementById('inline-edit-title').textContent=(type==='client'?'Rename Client':'Rename Project');
  document.getElementById('inline-edit-label').textContent='New Name';
  if(type==='client'){
    document.getElementById('inline-edit-value').value=val;
  } else {
    var proj=S.projects.find(function(p){return p.id===val;});
    document.getElementById('inline-edit-value').value=proj?proj.name:'';
  }
  document.getElementById('inline-edit-modal').classList.add('open');
}
function closeInlineEditModal(){document.getElementById('inline-edit-modal').classList.remove('open');}
function saveInlineEdit(){
  var newVal=document.getElementById('inline-edit-value').value.trim();
  if(!newVal){toast('Name cannot be empty');return;}
  if(S.inlineEditType==='client'){
    var old=S.inlineEditOldVal;
    var idx=S.clients.indexOf(old);
    if(idx>-1) S.clients[idx]=newVal;
    // Update entries
    S.entries.forEach(function(e){if(e.client===old)e.client=newVal;});
    S.projects.forEach(function(p){if(p.clientId===old)p.clientId=newVal;});
    S.expenses.forEach(function(e){if(e.client===old)e.client=newVal;});
    // Update user clientId
    USERS.forEach(function(u){if(u.clientId===old)u.clientId=newVal;});
    toast('Client renamed to "'+newVal+'"');
  } else {
    var proj=S.projects.find(function(p){return p.id===S.inlineEditOldVal;});
    if(proj){
      var old=proj.name;
      proj.name=newVal;
      S.entries.forEach(function(e){if(e.project===old)e.project=newVal;});
      toast('Project renamed to "'+newVal+'"');
    }
  }
  closeInlineEditModal();renderManageLists();populateSelects();renderAll();
}

function saveManualEntry(){
  var client=document.getElementById('m-client').value;
  var project=document.getElementById('m-project').value;
  var task=document.getElementById('m-task').value;
  var desc=document.getElementById('m-desc').value.trim();
  var date=document.getElementById('m-date').value;
  var start=document.getElementById('m-start').value;
  var end=document.getElementById('m-end').value;
  if(!date){toast('Please select a date');return;}
  if(!start||!end){toast('Please set start and end times');return;}
  var sh=parseInt(start.split(':')[0]),sm=parseInt(start.split(':')[1]);
  var eh=parseInt(end.split(':')[0]),em=parseInt(end.split(':')[1]);
  var hours=((eh*60+em)-(sh*60+sm))/60;
  if(hours<=0){toast('End time must be after start time');return;}
  if(S.editingEntryId){
    var idx=S.entries.findIndex(function(e){return e.id===S.editingEntryId;});
    if(idx>-1)S.entries[idx]=Object.assign({},S.entries[idx],{client:client,project:project,task:task,description:desc,date:date,start:start,end:end,hours:hours});
    toast('Entry updated ✓');
  } else {
    S.entries.push({id:Date.now(),userId:S.currentUser.id,date:date,client:client,project:project,task:task,description:desc,hours:hours,start:start,end:end,approvalStatus:'pending'});
    toast('Entry saved ✓');
  }
  closeManualModal();renderAll();
}

function setReportType(type,el){S.reportType=type;document.querySelectorAll('.report-tab').forEach(function(t){t.classList.remove('active');});el.classList.add('active');setReportDates();}
function setReportDates(){
  var now=new Date(),from=document.getElementById('report-from'),to=document.getElementById('report-to');
  if(S.reportType==='weekly'){
    var dow=now.getDay(),mon=new Date(now);mon.setDate(now.getDate()-(dow===0?6:dow-1));
    var sun=new Date(mon);sun.setDate(mon.getDate()+6);from.value=fmt(mon);to.value=fmt(sun);
  } else {
    from.value=fmt(new Date(now.getFullYear(),now.getMonth(),1));
    to.value=fmt(new Date(now.getFullYear(),now.getMonth()+1,0));
  }
}
function generateReport(){
  var from=document.getElementById('report-from').value;
  var to=document.getElementById('report-to').value;
  var cf=document.getElementById('report-client-filter').value;
  var ef=document.getElementById('report-emp-filter')?document.getElementById('report-emp-filter').value:'';
  var p=perm();
  var u=S.currentUser;
  var filtered=S.entries.filter(function(e){
    if(e.date<from||e.date>to)return false;
    if(cf&&e.client!==cf)return false;
    if(p.isClient&&e.client!==u.clientId)return false;
    if((p.canAdmin||p.canManage)&&ef&&e.userId!==ef)return false;
    if(!p.canAdmin&&!p.canManage&&!p.canApprove&&!p.isClient&&e.userId!==u.id)return false;
    return true;
  });
  var out=document.getElementById('report-output');
  if(!filtered.length){out.innerHTML='<div class="empty-state"><div class="empty-icon">📊</div><div class="empty-title">No data for this period</div><div class="empty-sub">Try adjusting filters or date range</div></div>';return;}
  var totalH=filtered.reduce(function(s,e){return s+e.hours;},0);
  var days=Math.max(1,Math.round((new Date(to)-new Date(from))/86400000)+1);
  var uniqC=[...new Set(filtered.map(function(e){return e.client;}).filter(Boolean))].length;
  var byClient={},byProject={},byEmployee={};
  filtered.forEach(function(e){
    byClient[e.client||'(No client)']=(byClient[e.client||'(No client)']||0)+e.hours;
    byProject[e.project||'(No project)']=(byProject[e.project||'(No project)']||0)+e.hours;
    var uname=(USERS.find(function(u){return u.id===e.userId;})||{name:'Unknown'}).name;
    byEmployee[uname]=(byEmployee[uname]||0)+e.hours;
  });
  function tbl(data,label){
    var rows=Object.entries(data).sort(function(a,b){return b[1]-a[1];});
    return '<div class="report-table"><div class="report-table-header" style="grid-template-columns:2fr 1fr 1fr;"><span>'+label+'</span><span class="text-right">Hours</span><span class="text-right">%</span></div>'+rows.map(function(x){return '<div class="report-row" style="grid-template-columns:2fr 1fr 1fr;"><span class="fw-600">'+x[0]+'</span><span class="text-right text-mono">'+fmtH(x[1])+'</span><span class="text-right text-mid">'+Math.round(x[1]/totalH*100)+'%</span></div>';}).join('')+'</div>';
  }
  var isElevated=p.canAdmin||p.canManage||p.canApprove;
  var stats='<div class="report-cards">'
    +'<div class="stat-card"><div class="stat-label">Total Hours</div><div class="stat-value">'+fmtH(totalH)+'</div><div class="stat-sub">'+from+' → '+to+'</div></div>'
    +'<div class="stat-card"><div class="stat-label">Entries</div><div class="stat-value">'+filtered.length+'</div></div>'
    +'<div class="stat-card"><div class="stat-label">Avg/Day</div><div class="stat-value">'+fmtH(totalH/days)+'</div></div>'
    +'<div class="stat-card"><div class="stat-label">Clients</div><div class="stat-value">'+uniqC+'</div></div>'
    +'</div>';
  var cols='<div style="display:grid;grid-template-columns:'+(isElevated?'1fr 1fr 1fr':'1fr 1fr')+';gap:16px;margin-bottom:20px;">';
  cols+='<div><div class="section-title" style="margin-bottom:8px;">By Client</div>'+tbl(byClient,'Client')+'</div>';
  cols+='<div><div class="section-title" style="margin-bottom:8px;">By Project</div>'+tbl(byProject,'Project')+'</div>';
  if(isElevated) cols+='<div><div class="section-title" style="margin-bottom:8px;">By Employee</div>'+tbl(byEmployee,'Employee')+'</div>';
  cols+='</div>';
  var gridCols=isElevated?'110px 1.2fr 1.2fr 1.5fr 1.2fr 70px':'110px 1.5fr 1.5fr 2fr 70px';
  var detail='<div class="section-title" style="margin-bottom:10px;">All Entries</div><div class="report-table"><div class="report-table-header" style="grid-template-columns:'+gridCols+'"><span>Date</span><span>Client</span><span>Project</span><span>Description</span>'+(isElevated?'<span>Employee</span>':'')+'<span class="text-right">Hours</span></div>'
    +filtered.sort(function(a,b){return a.date.localeCompare(b.date);}).map(function(e){
      var uname=(USERS.find(function(u){return u.id===e.userId;})||{name:'—'}).name;
      return '<div class="report-row" style="grid-template-columns:'+gridCols+'"><span class="text-mid" style="font-size:12px;">'+e.date+'</span><span class="fw-600" style="color:var(--accent)">'+(e.client||'—')+'</span><span>'+(e.project||'—')+'</span><span class="text-mid" style="font-size:12px;">'+(e.description||'')+'</span>'+(isElevated?'<span class="text-mid" style="font-size:12px;">'+uname+'</span>':'')+'<span class="text-right text-mono fw-600">'+fmtH(e.hours)+'</span></div>';
    }).join('')+'</div>';

  for(var i=0;i<totalCells;i++){
    var day,dateStr,isOther=false;
    if(i<startOffset){day=prevDays-startOffset+i+1;dateStr=fmt(new Date(y,m-1,day));isOther=true;}
    else if(i>=startOffset+daysInMonth){day=i-startOffset-daysInMonth+1;dateStr=fmt(new Date(y,m+1,day));isOther=true;}
    else{day=i-startOffset+1;dateStr=y+'-'+String(m+1).padStart(2,'0')+'-'+String(day).padStart(2,'0');}
    var isToday=dateStr===todayStr;
    var dayEvents=S.calEvents.filter(function(e){return e.date===dateStr;});
    var dayEntries=S.entries.filter(function(e){return e.date===dateStr&&(perm().canAdmin||perm().canManage||e.userId===S.currentUser.id);});
    html+='<div class="cal-day'+(isOther?' other-month':'')+(dayEntries.length?' has-entries':'')+'" onclick="openCalModal(\''+dateStr+'\')">';
    html+='<div class="cal-day-num" style="'+(isToday?'background:var(--accent);color:#fff;border-radius:50%;width:26px;height:26px;display:flex;align-items:center;justify-content:center;':'')+'" >'+day+'</div>';
    dayEntries.slice(0,2).forEach(function(e){html+='<div class="cal-event" style="background:var(--accent);">⏱ '+(e.project||e.client||'Entry').substring(0,14)+'</div>';});
    if(dayEntries.length>2)html+='<div style="font-size:10px;color:var(--ink-light);">+'+(dayEntries.length-2)+' more</div>';
    dayEvents.forEach(function(ev){html+='<div class="cal-event" style="background:'+ev.color+';" onclick="event.stopPropagation();deleteCalEvent('+ev.id+')" title="Click to delete">'+ev.title.substring(0,16)+'</div>';});
    html+='</div>';
  }
  document.getElementById('cal-grid').innerHTML=html;
}

function newNote(){var note={id:Date.now(),title:'Untitled Note',content:'',created:new Date().toISOString(),updated:new Date().toISOString()};S.notes.unshift(note);openNote(note.id);renderNoteList();}
function openNote(id){if(S.noteDirty&&S.currentNoteId)saveCurrentNote();S.currentNoteId=id;var note=S.notes.find(function(n){return n.id===id;});if(!note)return;document.getElementById('notebook-empty').style.display='none';document.getElementById('notebook-editor').style.display='flex';document.getElementById('note-title-input').value=note.title;document.getElementById('note-content-editor').innerHTML=note.content||'';S.noteDirty=false;renderNoteList();}
function saveCurrentNote(){if(!S.currentNoteId)return;var note=S.notes.find(function(n){return n.id===S.currentNoteId;});if(!note)return;note.title=document.getElementById('note-title-input').value.trim()||'Untitled Note';note.content=document.getElementById('note-content-editor').innerHTML;note.updated=new Date().toISOString();S.noteDirty=false;renderNoteList();toast('Note saved ✓');}
function deleteCurrentNote(){if(!S.currentNoteId)return;S.notes=S.notes.filter(function(n){return n.id!==S.currentNoteId;});S.currentNoteId=null;S.noteDirty=false;document.getElementById('notebook-empty').style.display='flex';document.getElementById('notebook-editor').style.display='none';renderNoteList();toast('Note deleted');}
function markNoteDirty(){S.noteDirty=true;}
function renderNoteList(){
  var q=(document.getElementById('note-search')||{}).value||'';
  var filtered=S.notes.filter(function(n){return !q||n.title.toLowerCase().indexOf(q.toLowerCase())>-1||(n.content||'').toLowerCase().indexOf(q.toLowerCase())>-1;});
  var el=document.getElementById('note-list');
  if(!filtered.length){el.innerHTML='<div style="text-align:center;color:var(--ink-light);font-size:13px;padding:20px;">No notes found</div>';return;}
  el.innerHTML=filtered.map(function(n){var preview=n.content?n.content.replace(/<[^>]+>/g,'').substring(0,60):'No content';var date=new Date(n.updated).toLocaleDateString('en-US',{month:'short',day:'numeric'});return '<div class="note-list-item'+(n.id===S.currentNoteId?' active':'')+'" onclick="openNote('+n.id+')"><div class="note-list-title">'+n.title+'</div><div class="note-list-preview">'+preview+'</div><div class="note-list-date">'+date+'</div></div>';}).join('');
}
